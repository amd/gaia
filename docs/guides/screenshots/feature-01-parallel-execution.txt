from gaia.orchestration import ProjectOrchestrator, OrchestratorConfig
import asyncio

async def demo():
    config = OrchestratorConfig(
        objectives_path='.gaia/objectives.yaml',
        enable_parallel_execution=True,
        max_parallel_objectives=4
    )
    orch = ProjectOrchestrator(config=config)
    
    # Level partitioning via Kahn's algorithm
    levels = orch.project.dependency_graph.partition_into_levels()
    print(f'Levels: {len(levels)}')
    for i, level in enumerate(levels):
        print(f'  Level {i}: {level}')

asyncio.run(demo())
