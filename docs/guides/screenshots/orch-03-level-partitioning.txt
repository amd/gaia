$ python -c "
from gaia.orchestration import ProjectOrchestrator
from gaia.orchestration.models import DependencyGraph, Objective, ProjectObjectives

objectives = [
    Objective(objective_id='obj-001', title='Initialize Project', phase='PLANNING'),
    Objective(objective_id='obj-002', title='Build Core Module', phase='DEVELOPMENT', dependencies=['obj-001']),
    Objective(objective_id='obj-003', title='Write Tests', phase='QUALITY', dependencies=['obj-002']),
    Objective(objective_id='obj-004', title='Design UI Components', phase='DEVELOPMENT', dependencies=['obj-001']),
    Objective(objective_id='obj-005', title='Integration Testing', phase='QUALITY', dependencies=['obj-002', 'obj-004']),
]
project = ProjectObjectives(project_id='demo-guide', objectives=objectives)
graph = DependencyGraph(objectives)
levels = graph.partition_into_levels()
"
